Swift Combine Network Test 코드 작성예제

2020년 02월 18일 Harry The Great

스위프트에서 비동기 코드를 테스트하기위해서는 XCTestExceptation을 사용합니다. XCTestExceptation은 호출이 완료될때까지 기다리게됩니다.

# 테스트코드 예제

class NetworkTestCase: XCTestCase {
    func testAPIList() {
        let expectation = XCTestExpectation(desc: self.description)
        let request = URLRequest("URL주소")
        let cancellable = URLSession.shared.dataTaskPublisher(for: request)
                .sink(receiveCompletion: { completion in
                    print("호출이 완료되었습니다.", String(describing: completion))
                    switch completion {
                    case .finished:
							//expectaion은 fullfill되면 종료됩니다.
                        expectation.fulfill()
                        break
                    case .failure(let networkError):
                        print("네트워크 처리중 에러가 발생하였습니다.", networkError)
                        XCTFail("No failure should be received from empty")
                        break
                    }
                }, receiveValue: { result in
                    XCTAssertNotNil(result)
                    print("데이터를 성공적으로 받았습니다. \(result)")
                })
        /* expectation의 호출이 완료될때까지 10초간 기다립니다.*/
        wait(for: [expectation], timeout: 10)
        XCTAssertNotNil(cancellable)
    }
}

위 코드는 테스트코들 실패를 만나게 되면 바로 종료됩니다. 만약 실패 코드를 작성하거나 다수의 네트워크 요청을 하는 테스트를 작성하려 할 때 종료되지 않고 테스트가 계속 되려 한다면 이때도 Expectation을 사용할 수 있습니다.

isInverted옵션
let failExcpetion = XCTestExpectation(desc: self.description)
//fullfill이 반대로 동작하도록합니다.
failExcpetion.isInverted = true

# 테스트헬퍼 사용하기

테스트 코드가 많아지면 그만큼 상용구 코드(Boiler Plate)도 많아지기 때문에 별도의 함수를 작성한다면 아래와 같이 작성할 수 있습니다.

final class NetworkTest: XCTestCase {


    func testResponseGetAPI_requestUser() throws {
        let publisher = URLSession.shared.dataTaskPublisher(for: request)
        let validTest = evaluateTestCode(publisher: publisher)
        wait(for: invalidTest.expectations, timeout: testTimeout)
        invalidTest.cancellable?.cancel()
    }

    func testResponsePostAPI_creatUser() throws {
        let publisher = URLSession.shared.dataTaskPublisher(for: request)
        let validTest = evaluateTestCode(publisher: publisher)
        wait(for: invalidTest.expectations, timeout: testTimeout)
        invalidTest.cancellable?.cancel()
    }


    /* 테스트 헬퍼 함수 */
    func evaluateTestCode<T: Publisher>(publisher: T?) -> (expectations: [XCTestExpectation], cancellable: AnyCancellable?) {

        /* Publisher를 체크합니다.*/
        XCTAssertNotNil(publisher)

        let expectationFinished = expectation(desc: "finished Test - \(self.description)")
        let expectationReceive = expectation(desc: "receiveValue  - \(self.description)")

        let cancellable = publisher?.sink(receiveCompletion: { (completion) in
            switch completion {
            case .failure(let error):
                /* 실패시 바로 끝나게됩니다. */
                XCTFail(error.localizedDescription)

            case .finished:
                expectationFinished.fulfill()
            }
        }, receiveValue: { response in
            //결과가 nil값이 아니어야합니다.
            XCTAssertNotNil(response)
            print(response)
            expectationReceive.fulfill()
        })
        //finish와 receive가 다 fullfill되어야 테스트를 마칩니다.
        return (expectations: [expectationFinished, expectationReceive], cancellable: cancellable)
    }
}
