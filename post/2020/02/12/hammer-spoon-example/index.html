# 특정앱 실행시키기

function launchApp(basicKey, object)
	//함수형태로 bindkey를 받습니다.
    hs.hotkey.bind(basicKey, object.key, function() 
		//매칭되는 어플리케이션을 실행시킵니다.
        hs.application.launchOrFocus(object.app)
        local application = hs.application.get(object.app)
        if application ~= nil then
            local window = application:focusedWindow()
            if window ~= nil then
                moveToCenterOfWindow(window)
            end
        end
    end)
end

# 현재 포커스되어있는 앱이름 가져오기

hs.hotkey.bind({'cmd'}, 'h', function()

    local focus  =  hs.application.frontmostApplication():name()
    if(focus == "Quiver") then
        print("gotcha!");
    end
    print(focus)

    -- print(focusAPP)
    local lay = hs.keycodes.currentLayout()
    if lay == "Unicode Hex Input" then
        hs.keycodes.setLayout("ABC")
    else
        hs.keycodes.setLayout("Unicode Hex Input")
    end
end)

# 어플리케션별로 바뀌는 키보드 레이아웃바꾸기

hs.hotkey.bind({'cmd'}, 'h', function()
    local unicodeLayout = "com.apple.keylayout.UnicodeHexInput"
    local gureumLayout = "org.youknowone.inputmethod.Gureum.han2"    
    local koreanLayout = "com.apple.inputmethod.Korean.2SetKorean"

    local focus  =  hs.application.frontmostApplication():name()
    local input_id =  hs.keycodes.currentSourceID()
    local promiseLayout

    if(input_id ~= unicodeLayout) then 
        promiseLayout = unicodeLayout
    else 
        if(focus == "Quiver") then
            promiseLayout = koreanLayout
        else
            promiseLayout = gureumLayout
        end
    end
    hs.keycodes.currentSourceID(promiseLayout)

end)
